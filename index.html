  <script>

    const venueLat = 14.6537; // Replace with your venue latitude

    const venueLon = 121.0685; // Replace with your venue longitude

    const allowedRadius = 0.1; // km (100 meters)

    const webhookURL = "https://hook.integromat.com/your_webhook_url"; // Replace with your Make webhook


    const form = document.getElementById('checkinForm');

    const status = document.getElementById('status');

    const retryBtn = document.getElementById('retry');


    form.addEventListener('submit', function(e) {

      e.preventDefault();

      startGPSCheck();

    });


    retryBtn.addEventListener('click', function() {

      retryBtn.style.display = "none";

      startGPSCheck();

    });


    function startGPSCheck() {

      const name = document.getElementById('name').value.trim();

      if (!name) {

        alert("Please enter your name.");

        return;

      }


      status.textContent = "Checking location...";

      if (navigator.geolocation) {

        navigator.geolocation.getCurrentPosition(success, fail, {enableHighAccuracy:true});

      } else {

        alert("Geolocation is not supported by your browser.");

      }


      function success(position) {

        const lat = position.coords.latitude;

        const lon = position.coords.longitude;

        const distance = getDistanceFromLatLonInKm(lat, lon, venueLat, venueLon);


        if (distance <= allowedRadius) {

          status.textContent = "Location verified! Redirecting...";

          // Send data to Make

          fetch(webhookURL, {

            method: 'POST',

            headers: {'Content-Type':'application/json'},

            body: JSON.stringify({name: name, latitude: lat, longitude: lon, timestamp: new Date().toISOString()})

          }).then(() => {

            window.location.href = "https://yoursite.com/resources"; // Replace with your resource page

          }).catch(() => {

            alert("Error sending data. Try again.");

          });

        } else {

          alert("You are not at the venue. Access denied.");

          retryBtn.style.display = "inline-block";

        }

      }


      function fail(error) {

        alert("GPS access is required. Please allow location and retry.");

        retryBtn.style.display = "inline-block";

        status.textContent = "";

      }

    }


    // Haversine formula

    function getDistanceFromLatLonInKm(lat1, lon1, lat2, lon2) {

      const R = 6371;

      const dLat = deg2rad(lat2-lat1);

      const dLon = deg2rad(lon2-lon1);

      const a = Math.sin(dLat/2)**2 + Math.cos(deg2rad(lat1))*Math.cos(deg2rad(lat2))*Math.sin(dLon/2)**2;

      const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

      return R*c;

    }


    function deg2rad(deg) {

      return deg * (Math.PI/180);

    }
